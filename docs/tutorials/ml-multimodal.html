Tutorial 1 — Classical ML baseline for molecular property prediction

Before reaching for deep learning, always establish a strong classical ML baseline.

from rdkit import Chem
from rdkit.Chem import AllChem, Descriptors, rdMolDescriptors
import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.model_selection import cross_val_score, StratifiedKFold
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import roc_auc_score

def get_features(smiles: str) -> np.ndarray:
    """Morgan fingerprint + physicochemical descriptors."""
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None
    fp   = AllChem.GetMorganFingerprintAsBitVect(mol, radius=2, nBits=1024)
    desc = [
        Descriptors.ExactMolWt(mol),
        Descriptors.MolLogP(mol),
        Descriptors.TPSA(mol),
        rdMolDescriptors.CalcNumHBD(mol),
        rdMolDescriptors.CalcNumHBA(mol),
        rdMolDescriptors.CalcNumRotatableBonds(mol),
        rdMolDescriptors.CalcNumAromaticRings(mol),
    ]
    return np.concatenate([list(fp), desc])

# Simulated hERG dataset (replace with real data from ChEMBL)
np.random.seed(42)
smiles_active   = ["CC(=O)Oc1ccccc1C(=O)O"] * 50   # placeholder
smiles_inactive = ["Cn1cnc2c1c(=O)n(C)c(=O)n2C"] * 50

X = np.array([get_features(s) for s in smiles_active + smiles_inactive])
y = np.array([1]*50 + [0]*50)

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

for name, clf in [("Random Forest",   RandomForestClassifier(n_estimators=100, random_state=42)),
                  ("Gradient Boost",  GradientBoostingClassifier(n_estimators=100, random_state=42))]:
    scores = cross_val_score(clf, X_scaled, y, cv=cv, scoring="roc_auc")
    print(f"{name:20s}  AUC = {scores.mean():.3f} ± {scores.std():.3f}")

Tutorial 2 — Graph Neural Network for molecular property prediction

Molecules are graphs — GNNs directly encode atoms (nodes) and bonds (edges).

Code 
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch_geometric.nn import GCNConv, global_mean_pool
from torch_geometric.data import Data, DataLoader
from rdkit import Chem
from rdkit.Chem import AllChem

def mol_to_graph(smiles: str, label: float) -> Data:
    """Convert SMILES to PyG graph."""
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None

    # Node features: atomic number, degree, formal charge, aromaticity
    x = []
    for atom in mol.GetAtoms():
        x.append([
            atom.GetAtomicNum(),
            atom.GetDegree(),
            atom.GetFormalCharge(),
            int(atom.GetIsAromatic()),
            atom.GetTotalNumHs(),
        ])
    x = torch.tensor(x, dtype=torch.float)

    # Edge indices (bonds)
    edge_index = []
    for bond in mol.GetBonds():
        i, j = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
        edge_index += [[i, j], [j, i]]
    edge_index = torch.tensor(edge_index, dtype=torch.long).t().contiguous()

    return Data(x=x, edge_index=edge_index, y=torch.tensor([label], dtype=torch.float))

class MolGCN(nn.Module):
    def __init__(self, in_channels=5, hidden=64):
        super().__init__()
        self.conv1 = GCNConv(in_channels, hidden)
        self.conv2 = GCNConv(hidden, hidden)
        self.conv3 = GCNConv(hidden, hidden)
        self.fc    = nn.Linear(hidden, 1)

    def forward(self, x, edge_index, batch):
        x = F.relu(self.conv1(x, edge_index))
        x = F.relu(self.conv2(x, edge_index))
        x = F.relu(self.conv3(x, edge_index))
        x = global_mean_pool(x, batch)   # graph-level embedding
        return torch.sigmoid(self.fc(x)).squeeze()

# Training loop
model     = MolGCN()
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
criterion = nn.BCELoss()

# (Replace with real DataLoader from your dataset)
for epoch in range(50):
    model.train()
    for batch in train_loader:
        optimizer.zero_grad()
        pred = model(batch.x, batch.edge_index, batch.batch)
        loss = criterion(pred, batch.y)
        loss.backward()
        optimizer.step()
    if epoch % 10 == 0:
        print(f"Epoch {epoch:3d}  loss={loss.item():.4f}")

Tutorial 3 — Transfer learning with ChemBERTa

Use a pre-trained molecular transformer (ChemBERTa) and fine-tune it on your toxicity dataset.

Code 
from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch
from torch.utils.data import Dataset, DataLoader

MODEL_NAME = "seyonec/ChemBERTa-zinc-base-v1"
tokenizer  = AutoTokenizer.from_pretrained(MODEL_NAME)
model      = AutoModelForSequenceClassification.from_pretrained(MODEL_NAME, num_labels=2)

class SmilesDataset(Dataset):
    def __init__(self, smiles_list, labels, tokenizer, max_len=128):
        self.encodings = tokenizer(smiles_list, truncation=True,
                                   padding=True, max_length=max_len,
                                   return_tensors="pt")
        self.labels = torch.tensor(labels, dtype=torch.long)

    def __len__(self):  return len(self.labels)
    def __getitem__(self, idx):
        return {k: v[idx] for k, v in self.encodings.items()}, self.labels[idx]

# Fine-tuning
smiles_train = ["CC(=O)Oc1ccccc1C(=O)O", "Cn1cnc2c1c(=O)n(C)c(=O)n2C"]
labels_train = [1, 0]

dataset    = SmilesDataset(smiles_train, labels_train, tokenizer)
loader     = DataLoader(dataset, batch_size=8, shuffle=True)
optimizer  = torch.optim.AdamW(model.parameters(), lr=2e-5)

model.train()
for epoch in range(3):
    for batch, labels in loader:
        outputs = model(**batch, labels=labels)
        loss    = outputs.loss
        loss.backward()
        optimizer.step()
        optimizer.zero_grad()
    print(f"Epoch {epoch+1}  loss={loss.item():.4f}")

model.save_pretrained("chemberta_toxicity_finetuned")
tokenizer.save_pretrained("chemberta_toxicity_finetuned")

## Tutorial 3 — Transfer learning with ChemBERTa

Use a pre-trained molecular transformer (ChemBERTa) and fine-tune it on your toxicity dataset.

from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch
from torch.utils.data import Dataset, DataLoader

MODEL_NAME = "seyonec/ChemBERTa-zinc-base-v1"
tokenizer  = AutoTokenizer.from_pretrained(MODEL_NAME)
model      = AutoModelForSequenceClassification.from_pretrained(MODEL_NAME, num_labels=2)

class SmilesDataset(Dataset):
    def __init__(self, smiles_list, labels, tokenizer, max_len=128):
        self.encodings = tokenizer(smiles_list, truncation=True,
                                   padding=True, max_length=max_len,
                                   return_tensors="pt")
        self.labels = torch.tensor(labels, dtype=torch.long)

    def __len__(self):  return len(self.labels)
    def __getitem__(self, idx):
        return {k: v[idx] for k, v in self.encodings.items()}, self.labels[idx]

# Fine-tuning
smiles_train = ["CC(=O)Oc1ccccc1C(=O)O", "Cn1cnc2c1c(=O)n(C)c(=O)n2C"]
labels_train = [1, 0]

dataset    = SmilesDataset(smiles_train, labels_train, tokenizer)
loader     = DataLoader(dataset, batch_size=8, shuffle=True)
optimizer  = torch.optim.AdamW(model.parameters(), lr=2e-5)

model.train()
for epoch in range(3):
    for batch, labels in loader:
        outputs = model(**batch, labels=labels)
        loss    = outputs.loss
        loss.backward()
        optimizer.step()
        optimizer.zero_grad()
    print(f"Epoch {epoch+1}  loss={loss.item():.4f}")

model.save_pretrained("chemberta_toxicity_finetuned")
tokenizer.save_pretrained("chemberta_toxicity_finetuned")
