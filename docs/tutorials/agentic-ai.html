Overview

These tutorials show how to build agentic AI systems — LLMs that can plan, use tools, and iterate — applied to real scientific workflows. We use LangChain, LangGraph, and the Anthropic/OpenAI APIs.

pip install langchain langgraph langchain-anthropic
pip install chromadb sentence-transformers

Tutorial 1 — A RAG pipeline over scientific literature

Build a Retrieval-Augmented Generation (RAG) system that answers drug-target questions by searching a corpus of your own papers and PubMed abstracts.

Code 
from langchain_community.vectorstores import Chroma
from langchain_community.embeddings import HuggingFaceEmbeddings
from langchain_anthropic import ChatAnthropic
from langchain.chains import RetrievalQA
from langchain.schema import Document
import requests, xml.etree.ElementTree as ET

# ── Step 1: Fetch PubMed abstracts ─────────────────────────────────────────
def fetch_pubmed(query: str, max_results: int = 20) -> list[Document]:
    base = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
    ids  = requests.get(f"{base}esearch.fcgi",
                        params={"db":"pubmed","term":query,
                                "retmax":max_results,"retmode":"json"}).json()
    pmids = ids["esearchresult"]["idlist"]
    xml   = requests.get(f"{base}efetch.fcgi",
                         params={"db":"pubmed","id":",".join(pmids),
                                 "rettype":"abstract","retmode":"xml"}).text
    root  = ET.fromstring(xml)
    docs  = []
    for art in root.iter("PubmedArticle"):
        title    = "".join(t.itertext() for t in art.iter("ArticleTitle"))
        abstract = "".join(t.itertext() for t in art.iter("AbstractText"))
        pmid     = art.findtext(".//PMID", "")
        if abstract:
            docs.append(Document(
                page_content=f"{title}\n\n{abstract}",
                metadata={"pmid": pmid, "title": title}
            ))
    return docs

# ── Step 2: Embed and index ─────────────────────────────────────────────────
docs      = fetch_pubmed("SILCS drug discovery protein-ligand binding")
embedder  = HuggingFaceEmbeddings(model_name="sentence-transformers/all-MiniLM-L6-v2")
vectordb  = Chroma.from_documents(docs, embedder, persist_directory="./pubmed_db")

# ── Step 3: RAG QA chain ────────────────────────────────────────────────────
llm = ChatAnthropic(model="claude-sonnet-4-20250514", temperature=0)
qa  = RetrievalQA.from_chain_type(
    llm=llm,
    retriever=vectordb.as_retriever(search_kwargs={"k": 4}),
    return_source_documents=True,
)

answer = qa.invoke("What is the SILCS method and how is it used in virtual screening?")
print(answer["result"])
print("\n--- Sources ---")
for doc in answer["source_documents"]:
    print(f"PMID {doc.metadata['pmid']}: {doc.metadata['title'][:80]}")

Tutorial 2 — A LangGraph agent that queries ChEMBL and PubChem

Build a multi-step agent that takes a compound name, finds its structure, retrieves bioactivity data from ChEMBL, and summarizes findings.

Code 
import requests
from langgraph.graph import StateGraph, END
from langchain_anthropic import ChatAnthropic
from langchain_core.tools import tool
from typing import TypedDict, Annotated
import operator

# ── Define tools ────────────────────────────────────────────────────────────
@tool
def get_smiles_from_name(compound_name: str) -> str:
    """Get canonical SMILES for a compound name using PubChem."""
    url = f"https://pubchem.ncbi.nlm.nih.gov/rest/pug/compound/name/{compound_name}/property/CanonicalSMILES/JSON"
    r = requests.get(url)
    if r.ok:
        return r.json()["PropertyTable"]["Properties"][0]["CanonicalSMILES"]
    return f"Could not find SMILES for {compound_name}"

@tool
def get_chembl_bioactivity(compound_name: str, target_keyword: str = "hERG") -> str:
    """Retrieve bioactivity data from ChEMBL for a compound against a target."""
    search_url = f"https://www.ebi.ac.uk/chembl/api/data/activity.json?molecule_pref_name__icontains={compound_name}&target_pref_name__icontains={target_keyword}&limit=5"
    r = requests.get(search_url)
    if r.ok:
        activities = r.json().get("activities", [])
        if not activities:
            return f"No ChEMBL activity found for {compound_name} vs {target_keyword}"
        lines = []
        for act in activities[:5]:
            lines.append(f"  IC50={act.get('standard_value')} {act.get('standard_units')} | assay={act.get('assay_description','N/A')[:60]}")
        return "\n".join(lines)
    return "ChEMBL API error"

# ── Agent state ─────────────────────────────────────────────────────────────
class AgentState(TypedDict):
    compound:    str
    smiles:      str
    bioactivity: str
    summary:     str
    messages:    Annotated[list, operator.add]

# ── Node functions ───────────────────────────────────────────────────────────
llm   = ChatAnthropic(model="claude-sonnet-4-20250514", temperature=0)
tools = [get_smiles_from_name, get_chembl_bioactivity]

def fetch_structure(state: AgentState) -> AgentState:
    smiles = get_smiles_from_name.invoke(state["compound"])
    return {"smiles": smiles}

def fetch_bioactivity(state: AgentState) -> AgentState:
    data = get_chembl_bioactivity.invoke({"compound_name": state["compound"]})
    return {"bioactivity": data}

def summarize(state: AgentState) -> AgentState:
    prompt = f"""Compound: {state['compound']}
SMILES: {state['smiles']}
ChEMBL hERG bioactivity:
{state['bioactivity']}

Write a 3-sentence summary for a medicinal chemist."""
    summary = llm.invoke(prompt).content
    return {"summary": summary}

# ── Build graph ─────────────────────────────────────────────────────────────
graph = StateGraph(AgentState)
graph.add_node("fetch_structure",  fetch_structure)
graph.add_node("fetch_bioactivity", fetch_bioactivity)
graph.add_node("summarize",        summarize)

graph.set_entry_point("fetch_structure")
graph.add_edge("fetch_structure",   "fetch_bioactivity")
graph.add_edge("fetch_bioactivity", "summarize")
graph.add_edge("summarize",         END)

app    = graph.compile()
result = app.invoke({"compound": "terfenadine", "messages": []})
print(result["summary"])
