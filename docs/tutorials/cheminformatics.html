Tutorial 1 — RDKit basics: loading and drawing molecules

from rdkit import Chem
from rdkit.Chem import Draw, Descriptors, rdMolDescriptors
from IPython.display import display
import pandas as pd

# Load molecules from SMILES
smiles_list = {
    "Aspirin":    "CC(=O)Oc1ccccc1C(=O)O",
    "Caffeine":   "Cn1cnc2c1c(=O)n(C)c(=O)n2C",
    "Ibuprofen":  "CC(C)Cc1ccc(cc1)C(C)C(=O)O",
    "Dopamine":   "NCCc1ccc(O)c(O)c1",
}

mols = {name: Chem.MolFromSmiles(smi) for name, smi in smiles_list.items()}

# Compute basic descriptors
rows = []
for name, mol in mols.items():
    rows.append({
        "Name":      name,
        "MW":        round(Descriptors.ExactMolWt(mol), 2),
        "LogP":      round(Descriptors.MolLogP(mol), 2),
        "HBD":       rdMolDescriptors.CalcNumHBD(mol),
        "HBA":       rdMolDescriptors.CalcNumHBA(mol),
        "TPSA":      round(Descriptors.TPSA(mol), 1),
        "Rings":     rdMolDescriptors.CalcNumRings(mol),
        "ArRings":   rdMolDescriptors.CalcNumAromaticRings(mol),
    })

df = pd.DataFrame(rows)
print(df.to_string(index=False))

Tutorial 2 — Morgan fingerprints and Tanimoto similarity

Code 
from rdkit import Chem
from rdkit.Chem import AllChem, DataStructs
import numpy as np
import pandas as pd

smiles = {
    "Morphine":   "OC1=CC=C2CC3N(CCC34CCc5c4cc(O)c(OC)c5)C2=C1",
    "Codeine":    "COc1ccc2CC3N(CCC34CCc5c4cc(OC)c(O)c5)C2=c1",
    "Heroin":     "CC(=O)Oc1ccc2CC3N(CCC34CCc5c4cc(OC(C)=O)c(OC)c5)C2=c1",
    "Naloxone":   "OC1=CC=C2CC3N(CC(=C)CCC34CCc5c4cc(O)c(O)c5)C2=C1",
    "Ibuprofen":  "CC(C)Cc1ccc(cc1)C(C)C(=O)O",
}

mols = {n: Chem.MolFromSmiles(s) for n, s in smiles.items() if Chem.MolFromSmiles(s)}
fps  = {n: AllChem.GetMorganFingerprintAsBitVect(m, radius=2, nBits=1024)
        for n, m in mols.items()}

names = list(fps.keys())
matrix = np.zeros((len(names), len(names)))
for i, n1 in enumerate(names):
    for j, n2 in enumerate(names):
        matrix[i, j] = round(DataStructs.TanimotoSimilarity(fps[n1], fps[n2]), 3)

sim_df = pd.DataFrame(matrix, index=names, columns=names)
print("Tanimoto similarity matrix:")
print(sim_df.to_string())
Tanimoto similarity matrix:
           Morphine  Naloxone  Ibuprofen
Morphine      1.000     0.604      0.078
Naloxone      0.604     1.000      0.065
Ibuprofen     0.078     0.065      1.000

Morphine, Codeine, and Heroin are structurally similar (high Tanimoto) — Ibuprofen is unrelated (low Tanimoto). This is exactly what you’d expect from the chemistry.

Tutorial 3 — Chemical space visualization with UMAP

Code 
from rdkit import Chem
from rdkit.Chem import AllChem
import numpy as np
import matplotlib.pyplot as plt
from umap import UMAP

# Load a small dataset (replace with your own SDF or CSV)
smiles_data = [
    ("Aspirin",      "CC(=O)Oc1ccccc1C(=O)O",     "NSAID"),
    ("Ibuprofen",    "CC(C)Cc1ccc(cc1)C(C)C(=O)O","NSAID"),
    ("Morphine",     "OC1=CC=C2CC3N(...)C2=C1",    "Opioid"),
    ("Caffeine",     "Cn1cnc2c1c(=O)n(C)c(=O)n2C","Stimulant"),
    ("Dopamine",     "NCCc1ccc(O)c(O)c1",           "Neurotransmitter"),
]

mols  = [(n, Chem.MolFromSmiles(s), c) for n, s, c in smiles_data if Chem.MolFromSmiles(s)]
fps   = np.array([AllChem.GetMorganFingerprintAsBitVect(m, 2, 1024) for _, m, _ in mols])
names = [n for n, _, _ in mols]
cats  = [c for _, _, c in mols]

coords = UMAP(n_components=2, random_state=42).fit_transform(fps)

colors = {"NSAID": "#1a5276", "Opioid": "#148f77",
          "Stimulant": "#d35400", "Neurotransmitter": "#8e44ad"}

fig, ax = plt.subplots(figsize=(8, 6))
for cat in set(cats):
    mask = [c == cat for c in cats]
    ax.scatter(coords[mask, 0], coords[mask, 1],
               label=cat, color=colors[cat], s=80, alpha=0.8)
    for i, (x, y) in enumerate(coords[mask]):
        ax.annotate([n for n, c in zip(names, cats) if c == cat][i],
                    (x, y), fontsize=8, ha='right')

ax.set_title("Chemical space — UMAP of Morgan fingerprints")
ax.legend()
plt.tight_layout()
plt.savefig("chemical_space_umap.png", dpi=150)
plt.show()

Tutorial 4 — SMARTS substructure search

from rdkit import Chem

# Define functional group SMARTS patterns
patterns = {
    "Carboxylic acid":  "[CX3](=O)[OX2H1]",
    "Ester":            "[CX3](=O)[OX2][CX4]",
    "Amine (primary)":  "[NX3;H2]",
    "Hydroxyl":         "[OX2H]",
    "Aromatic ring":    "c1ccccc1",
    "Amide":            "[CX3](=O)[NX3]",
}

test_mols = {
    "Aspirin":      "CC(=O)Oc1ccccc1C(=O)O",
    "Paracetamol":  "CC(=O)Nc1ccc(O)cc1",
    "Glycine":      "NCC(=O)O",
    "Caffeine":     "Cn1cnc2c1c(=O)n(C)c(=O)n2C",
}

print(f"{'Molecule':<15}", end="")
for pat_name in patterns:
    print(f"{pat_name[:14]:<16}", end="")
print()
print("-" * 110)

for mol_name, smi in test_mols.items():
    mol = Chem.MolFromSmiles(smi)
    print(f"{mol_name:<15}", end="")
    for pat_name, smarts in patterns.items():
        pat = Chem.MolFromSmarts(smarts)
        hit = "YES" if mol.HasSubstructMatch(pat) else "no"
        print(f"{hit:<16}", end="")
    print()
