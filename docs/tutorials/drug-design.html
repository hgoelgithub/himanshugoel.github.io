Tutorial 1 — Molecular fingerprints and similarity search

A hands-on introduction to molecular representation and similarity searching using RDKit — the foundation of cheminformatics and CADD.

from rdkit import Chem
from rdkit.Chem import AllChem, DataStructs, Draw
from rdkit.Chem import rdMolDescriptors
import pandas as pd

# Example: compare known hERG blockers vs non-blockers
smiles_dict = {
    "Terfenadine (blocker)":  "OC(c1ccc(C(c2ccccc2)(c2ccccc2)O)cc1)CCCN1CCC(CC1)C(O)(c1ccccc1)c1ccccc1",
    "Cisapride (blocker)":    "CCOC(=O)c1cc2cc(OC)c(OC)cc2[nH]1",
    "Metformin (non-blocker)":"CN(C)C(=N)NC(=N)N",
    "Caffeine (non-blocker)": "Cn1cnc2c1c(=O)n(C)c(=O)n2C",
}

mols = {name: Chem.MolFromSmiles(smi) for name, smi in smiles_dict.items()}
fps  = {name: AllChem.GetMorganFingerprintAsBitVect(mol, radius=2, nBits=2048)
        for name, mol in mols.items() if mol}

# Tanimoto similarity matrix
rows = []
for n1, fp1 in fps.items():
    for n2, fp2 in fps.items():
        rows.append({
            "Mol A": n1.split(" ")[0],
            "Mol B": n2.split(" ")[0],
            "Tanimoto": round(DataStructs.TanimotoSimilarity(fp1, fp2), 3)
        })

sim_df = pd.DataFrame(rows)
pivot  = sim_df.pivot(index="Mol A", columns="Mol B", values="Tanimoto")
print(pivot.to_string())
Mol B        Caffeine  Cisapride  Metformin  Terfenadine
Mol A                                                   
Caffeine        1.000      0.074      0.024        0.063
Cisapride       0.074      1.000      0.042        0.056
Metformin       0.024      0.042      1.000        0.017
Terfenadine     0.063      0.056      0.017        1.000

Key observations: - Terfenadine and Cisapride (both hERG blockers) show low Tanimoto similarity — toxicity isn’t structurally conserved - This motivates feature-based ML over pure similarity search for safety prediction - Morgan fingerprints (ECFP) encode local chemical environments at radius=2 (ECFP4)

Tutorial 2 — Lipinski’s Rule of Five and ADMET profiling

Screen a compound library for drug-likeness before committing to expensive docking runs.

Code 
from rdkit import Chem
from rdkit.Chem import Descriptors, rdMolDescriptors
import pandas as pd

def compute_ro5(smiles: str, name: str = "") -> dict:
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return {}
    mw   = Descriptors.ExactMolWt(mol)
    logp = Descriptors.MolLogP(mol)
    hbd  = rdMolDescriptors.CalcNumHBD(mol)
    hba  = rdMolDescriptors.CalcNumHBA(mol)
    tpsa = Descriptors.TPSA(mol)
    rot  = rdMolDescriptors.CalcNumRotatableBonds(mol)
    pass_ro5 = (mw <= 500) and (logp <= 5) and (hbd <= 5) and (hba <= 10)
    return {
        "Name": name, "MW": round(mw, 1), "LogP": round(logp, 2),
        "HBD": hbd, "HBA": hba, "TPSA": round(tpsa, 1),
        "RotBonds": rot, "Pass Ro5": "✅" if pass_ro5 else "❌"
    }

# Example library
library = {
    "Aspirin":       "CC(=O)Oc1ccccc1C(=O)O",
    "Ibuprofen":     "CC(C)Cc1ccc(cc1)C(C)C(=O)O",
    "Atorvastatin":  "CC(C)c1c(C(=O)Nc2ccccc2)c(-c2ccccc2)c(-c2ccc(F)cc2)n1CCC(O)CC(O)CC(=O)O",
    "Paclitaxel":    "O=C(Nc1ccccc1)c1ccccc1OC(=O)[C@@H]2[C@@H](OC(C)(C)C(=O)O)CC[C@@]3(O)C(=O)[C@H](OC(=O)c4ccccc4)[C@@H](CC[C@H]2OC(C)=O)O3",
    "Cyclosporin A": "CCC1NC(=O)C(C(O)C(C)CC=CC)N(C)C(=O)C(C(C)CC)N(C)C(=O)C(CC(C)C)N(C)C(=O)C(CC(C)C)N(C)C(=O)C(C)N(C)C(=O)C(CC(C)C)N(C)C(=O)C(CC(C)C)N(C)C(=O)C(C)N(C)C(=O)CN(C)C1=O",
}

results = pd.DataFrame([compute_ro5(smi, name) for name, smi in library.items()])
print(results.to_string(index=False))

Tutorial 3 — Virtual screening with AutoDock Vina (Python wrapper)

Set up a basic docking experiment against a receptor from the PDB.

Code 
import subprocess, os
from rdkit import Chem
from rdkit.Chem import AllChem

def prepare_ligand(smiles: str, out_pdbqt: str):
    """Convert SMILES → 3D → PDBQT for Vina."""
    mol = Chem.MolFromSmiles(smiles)
    mol = Chem.AddHs(mol)
    AllChem.EmbedMolecule(mol, AllChem.ETKDGv3())
    AllChem.MMFFOptimizeMolecule(mol)
    Chem.MolToMolFile(mol, out_pdbqt.replace(".pdbqt", ".mol"))
    # Convert with Open Babel (must be installed)
    subprocess.run(["obabel", out_pdbqt.replace(".pdbqt", ".mol"),
                    "-O", out_pdbqt, "--gen3d"], check=True)

def run_vina(receptor_pdbqt: str, ligand_pdbqt: str,
             center: tuple, box_size: tuple = (20, 20, 20),
             exhaustiveness: int = 8) -> float:
    """Run Vina and return best docking score (kcal/mol)."""
    cmd = [
        "vina",
        "--receptor", receptor_pdbqt,
        "--ligand",   ligand_pdbqt,
        "--center_x", str(center[0]),
        "--center_y", str(center[1]),
        "--center_z", str(center[2]),
        "--size_x",   str(box_size[0]),
        "--size_y",   str(box_size[1]),
        "--size_z",   str(box_size[2]),
        "--exhaustiveness", str(exhaustiveness),
        "--out", ligand_pdbqt.replace(".pdbqt", "_out.pdbqt"),
    ]
    result = subprocess.run(cmd, capture_output=True, text=True)
    for line in result.stdout.split("\n"):
        if line.strip().startswith("1"):
            return float(line.split()[1])
    return 0.0

# Example: dock a small library against SARS-CoV-2 main protease (PDB: 6LU7)
# center_x/y/z from known active site
library_smiles = {
    "Nirmatrelvir": "CC1(C2CC2CN1C(=O)C(NC(=O)C3CC3NC(=O)C(C#N)=C)C(C)(C)C)F",
    "GC-376":       "CC(C)(C)OC(=O)NC1CC(=O)N2CCCC12",
    "Ebselen":      "O=C1OC(c2ccccc21)=Nc1ccccc1[Se]1",
}

results = {}
for name, smi in library_smiles.items():
    lig_pdbqt = f"/tmp/{name.replace(' ','_')}.pdbqt"
    prepare_ligand(smi, lig_pdbqt)
    score = run_vina("6lu7_receptor.pdbqt", lig_pdbqt,
                     center=(-10.4, 13.1, 59.1))
    results[name] = score
    print(f"{name:25s}  {score:6.2f} kcal/mol")

This tutorial requires AutoDock Vina and Open Babel installed.
See the GitHub repo for a ready-to-run Docker container.
