Tutorial 1 — RNA-Seq QC and preprocessing

A well-structured RNA-Seq pipeline starts with quality checks before any analysis.

Code 
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Simulate a count matrix (genes x samples)
np.random.seed(42)
n_genes, n_samples = 500, 6
gene_names   = [f"Gene_{i:04d}" for i in range(n_genes)]
sample_names = ["Control_1","Control_2","Control_3","Treated_1","Treated_2","Treated_3"]

# Simulate counts: treated samples have ~20% genes upregulated
counts = np.random.negative_binomial(5, 0.3, size=(n_genes, n_samples))
counts[50:100, 3:] = (counts[50:100, 3:] * 4).astype(int)  # upregulated
counts[100:130, 3:] = (counts[100:130, 3:] * 0.2).astype(int)  # downregulated

count_df = pd.DataFrame(counts, index=gene_names, columns=sample_names)

# Basic QC: library sizes
lib_sizes = count_df.sum()
print("Library sizes (total counts per sample):")
for s, n in lib_sizes.items():
    print(f"  {s}: {n:,}")

# Flag low-count genes
low_count = (count_df < 10).all(axis=1).sum()
print(f"\nGenes with counts < 10 in ALL samples: {low_count} ({low_count/n_genes*100:.1f}%) — will be filtered")

# After filtering
filtered = count_df[count_df.sum(axis=1) >= 10]
print(f"Genes remaining after filter: {len(filtered)}")

Tutorial 2 — Differential expression with PyDESeq2

Code 
import pandas as pd
import numpy as np
from pydeseq2.dds import DeseqDataSet
from pydeseq2.ds import DeseqStats

# Load your count matrix (genes as rows, samples as columns)
counts = pd.read_csv("counts.csv", index_col=0)

# Sample metadata
metadata = pd.DataFrame({
    "condition": ["control","control","control","treated","treated","treated"]
}, index=counts.columns)

# Build DESeq2 dataset
dds = DeseqDataSet(
    counts=counts.T,           # PyDESeq2 expects samples as rows
    metadata=metadata,
    design_factors="condition",
    refit_cooks=True,
)
dds.deseq2()

# Run Wald test
stat_res = DeseqStats(dds, contrast=["condition", "treated", "control"])
stat_res.summary()

results = stat_res.results_df
results = results.dropna().sort_values("padj")

# Filter significant DEGs
sig = results[(results["padj"] < 0.05) & (results["log2FoldChange"].abs() > 1)]
print(f"Significant DEGs (padj<0.05, |LFC|>1): {len(sig)}")
print(f"  Upregulated:   {(sig['log2FoldChange'] > 0).sum()}")
print(f"  Downregulated: {(sig['log2FoldChange'] < 0).sum()}")
print(sig.head(10)[["baseMean","log2FoldChange","pvalue","padj"]])

Tutorial 3 — Volcano plot visualization

Code 
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches

# Simulate DESeq2-like results
np.random.seed(0)
n = 1000
log2fc = np.random.normal(0, 1.5, n)
pval   = np.random.uniform(0, 1, n)
# Make some genes truly significant
log2fc[50:80]  =  np.random.uniform(2, 5, 30)
pval[50:80]    =  np.random.uniform(1e-10, 1e-4, 30)
log2fc[80:100] = -np.random.uniform(2, 5, 20)
pval[80:100]   =  np.random.uniform(1e-8, 1e-3, 20)

neg_log10_p = -np.log10(pval + 1e-300)
padj_thresh = 0.05
lfc_thresh  = 1.0

up   = (log2fc >  lfc_thresh) & (pval < padj_thresh)
down = (log2fc < -lfc_thresh) & (pval < padj_thresh)
ns   = ~up & ~down

fig, ax = plt.subplots(figsize=(8, 6))
ax.scatter(log2fc[ns],   neg_log10_p[ns],   c="#b0b0b0", s=12, alpha=0.5, label=f"NS ({ns.sum()})")
ax.scatter(log2fc[up],   neg_log10_p[up],   c="#c0392b", s=20, alpha=0.8, label=f"Up ({up.sum()})")
ax.scatter(log2fc[down], neg_log10_p[down], c="#2980b9", s=20, alpha=0.8, label=f"Down ({down.sum()})")

ax.axhline(-np.log10(padj_thresh), color="black", linestyle="--", linewidth=0.8)
ax.axvline( lfc_thresh,  color="gray", linestyle="--", linewidth=0.8)
ax.axvline(-lfc_thresh,  color="gray", linestyle="--", linewidth=0.8)

ax.set_xlabel("log2 Fold Change", fontsize=12)
ax.set_ylabel("-log10(p-value)", fontsize=12)
ax.set_title("Volcano plot — Treated vs Control", fontsize=13)
ax.legend(frameon=False)
plt.tight_layout()
plt.savefig("volcano_plot.png", dpi=150)
plt.show()

Tutorial 4 — In vitro–in vivo correlation (IVIVE)

A key part of translational toxicology — correlating gene expression changes between animal models and human cell systems.

Code 
import numpy as np
import matplotlib.pyplot as plt
from scipy import stats

# Simulate log2FC values for the same genes in two systems
np.random.seed(7)
n_genes = 200
# Good IVIVE: most genes correlate, some scatter
invivo_lfc  = np.random.normal(0, 2, n_genes)
invitro_lfc = invivo_lfc * 0.75 + np.random.normal(0, 0.8, n_genes)

r, pval = stats.pearsonr(invivo_lfc, invitro_lfc)

fig, ax = plt.subplots(figsize=(6, 6))
ax.scatter(invivo_lfc, invitro_lfc, alpha=0.5, s=20, color="#1a5276")

# Regression line
m, b = np.polyfit(invivo_lfc, invitro_lfc, 1)
x_line = np.linspace(invivo_lfc.min(), invivo_lfc.max(), 100)
ax.plot(x_line, m * x_line + b, color="#d35400", linewidth=1.5)

ax.set_xlabel("In vivo log2FC (rat)", fontsize=12)
ax.set_ylabel("In vitro log2FC (human hepatocytes)", fontsize=12)
ax.set_title(f"IVIVE correlation  r = {r:.3f}, p = {pval:.2e}", fontsize=12)
ax.axhline(0, color="gray", linewidth=0.5)
ax.axvline(0, color="gray", linewidth=0.5)
plt.tight_layout()
plt.savefig("ivive_correlation.png", dpi=150)
plt.show()
